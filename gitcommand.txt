git init //初始化仓库
git config //获取配置信息
git help command //联网获取帮助
git status //查看当前工作目录的状态
git clone url  //克隆远程仓库到本地
git add --all //可选择文件夹，所有文件，单个文件
git commit -m "log" //提交到本地仓库，必须添加日志
git diff //查看本地仓库的与正在修改的不同的地方
git push origin master // 推送到远程仓库 origin为远程仓库.master为本地仓库
git checkout origin //撤销操作
git rm file //删除本地文件
git rm file -f //强制删除本地文件
git pull //将远程仓库的文件更新到本地
git log //查看日志
git log -p -2 //限制查看两条日志
git remote //查看所有的远程仓库, 加上-v的话，显示url
git fetch origin //抓取远程仓库的内容
git merge to local //将fetch到的内容合并到本地
git pull //等于fetch + merge to local
git tag // 添加标签
git branch name //创建一个名为name的分支
git checkout name // 将当前分支改为name分支
git log --oneline --decorate //显示所有分支，并显示当前分支的信息
git reset HEAD file //将文件从暂存区移除
git show //显示当前分支的状态
